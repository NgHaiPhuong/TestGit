    public static boolean isPrimeNumber(int n) {
    	// so nguyen n < 2 khong phai la so nguyen to
	    if (n < 2) {
	        return false;
	    }
        // check so nguyen to khi n >= 2
       int s = (int) Math.sqrt(n);
          for (int i = 2 ; i <= s ; i++) {
	           if (n % i == 0) {
	                return false;
	           }
	       }
	       return true;
    }
